Module type Game.Square

module type Square = sig .. end
type t

The abstract type of a square on a Chess Board.

val build : Game.position -> ChessPiece.t option -> int -> int -> t

build pos piece_opt w_thr b_thr creates a new square with position pos, item piece_opt, and is threatened by int w_thr white pieces and int b_thr black pieces.

val build_with_piece : Game.position -> Game.name -> Game.player -> int -> int -> t

build_with_piece pos name player w_thr b_thr creates a new square with position pos, piece with the same position, name name, and player player, and is threatened by int w_thr white pieces and int b_thr black pieces.

val get_position : t -> Game.position

the coordinates of the square t on the chessboard

val get_piece : t -> ChessPiece.t option

get_piece returns the chess piece that is on the square, and None of the square is empty

val get_wthreat : t -> int

get_wthreat is the amount of white pieces on the board that can legally move to this square

val get_bthreat : t -> int

get_bthreat is the amount of white pieces on the board that can legally move to this square
