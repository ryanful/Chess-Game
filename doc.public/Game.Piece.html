val build : Game.name -> Game.position -> Game.player -> t

build name position player is piece p' with name n, position p, and owned by player p.

val get_name : t -> Game.name

get_name piece is the name of piece

val get_player : t -> Game.player

get_player piece is the player that owns piece.

val get_position : t -> Game.position

get_position piece is the current location of piece

val move : t -> Game.position -> t

move p pos is piece p', which is piece p with its position changed to position pos

val is_players_piece : Game.player -> t option -> bool

is_player_piece player p is true if the player associated with piece p matches player. Otherwise false.

val player_to_string : Game.player -> string

player_to_string pl is the string representation of player pl.

val to_string : t -> string

to_string p is the string representation of piece p.
